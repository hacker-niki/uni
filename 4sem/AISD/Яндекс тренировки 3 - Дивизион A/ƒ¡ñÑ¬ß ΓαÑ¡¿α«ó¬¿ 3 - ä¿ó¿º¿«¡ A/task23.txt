#include <iostream>

long long T(long long n)
{
    return n * (n + 1) / 2;
}

long long mas[1000001][2];

int main()
{
    long long n;
    std::cin >> n;

    mas[1][1] = 0;
    mas[2][1] = 1;
    mas[1][0] = 1;
    mas[2][0] = 3;

    for (int i = 2; i < n + 1; i++)
    {
        mas[i][0] = mas[i - 1][0] + T(i);
        mas[i][1] = mas[i - 2][1] + T(i - 1);
    }

    std::cout << mas[n][0] + mas[n][1];
    return 0;
}
