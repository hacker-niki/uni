bool checkXML(std::string str)
{
    std::stack<std::string> st;
    std::string strtost = "";
    bool flag = true;
    //int back = 0;                        <a>>/a>
    try
    {
        if (str[0] != '<')
        {
            return false;
        }
        for (int i = 0; i < str.length(); i++)
        {
            if (str[i] == '>' && ((i != str.length() - 1 && str[i + 1] != '<') || str[i - 1] == '>' || str[i - 1] == '<' || str[i - 1] == '/'))
                return false;
            if (str[i] == '<' && (str[i + 1] == '<' || str[i + 1] == '>' || (i != 0 && str[i - 1] != '>')))
                return false;
            if (str[i] == '/' && str[i - 1] != '<')
                return false;
            if (str[i] == '<')
            {
                flag = true;
            }
            else if (str[i] == '>')
            {
                if (flag)
                {
                    st.push(strtost);
                }
                else if (!st.empty() and st.top() == strtost)
                {
                    st.pop();
                }
                else
                {
                    return false;
                }
                strtost = "";
            }
            else if (str[i] == '/')
            {
                flag = false;
            }
            else
            {
                strtost += str[i];
            }
        }
        if (st.empty())
        {
            return true;
        }
        else
        {
            return false;
        }
    }
    catch (...)
    {
        return false;
    }
}
